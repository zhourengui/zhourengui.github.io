工具模块

前言：

在 Node.js 模块库中有很多好用的模块。接下来我们为大家介绍几种常用模块的使用：OS、Path、Net、DNS、Domain。

OS

const OS = require("os")

// 返回操作系统的默认临时文件夹。
OS.tmpdir() // /var/folders/kl/f7kr9kds7fddc5j2tz6mh51h0000gn/T
// 返回 CPU 的字节序，可能的是 "BE" 或 "LE"。
OS.endianness() // LE
// 返回操作系统的主机名。
OS.hostname() // LittleBoydeMacBook-Pro.local
// 返回操作系统名
OS.type() // Darwin
// 返回编译时的操作系统名
OS.platform() // darwin
// 返回操作系统 CPU 架构，可能的值有 "x64"、"arm" 和 "ia32"。
OS.arch() // x64
// 返回操作系统的发行版本。
OS.release() // 18.7.0
// 返回操作系统运行的时间，以秒为单位。
OS.uptime() // 11296
// 返回一个包含 1、5、15 分钟平均负载的数组。
OS.loadavg() // [ 3.28515625, 2.6826171875, 2.4609375 ]
// 返回系统内存总量，单位为字节。
OS.totalmem() // 17179869184
// 返回操作系统空闲内存量，单位是字节。
OS.freemem() // 983805952
// 返回一个对象数组，包含所安装的每个 CPU/内核的信息：型号、速度（单位 MHz）、时间（一个包含 user、nice、sys、idle 和 irq 所使用 CPU/内核毫秒数的对象）。
OS.cpus()
// 获得网络接口列表。
OS.networkInterfaces()

Path

const path = require("path")

// 规范化路径，注意'..' 和 '.'。
path.normalize(".//test/test") // test/test
// 用于连接路径。该方法的主要用途在于，会正确使用当前系统的路径分隔符，Unix系统是"/"，Windows系统是"\"。
path.join("./test", "/test1") // test/test1
// 将 to 参数解析为绝对路径，给定的路径的序列是从右往左被处理的，后面每个 path 被依次解析，直到构造完成一个绝对路径。 例如，给定的路径片段的序列为：/foo、/bar、baz，则调用 path.resolve('/foo', '/bar', 'baz') 会返回 /bar/baz。
path.resolve("test", "test1", "test2") // /Users/littleboy/Desktop/node-test/test/test1/test2
// 判断参数 path 是否是绝对路径。
path.isAbsolute("./test") // false
// 用于将绝对路径转为相对路径，返回从 from 到 to 的相对路径（基于当前工作目录）。
path.relative("/", "test1") // Users/littleboy/Desktop/node-test/test1
// 返回路径中代表文件夹的部分，同 Unix 的dirname 命令类似。
path.dirname("./test") // .
// 返回路径中的最后一部分。同 Unix 命令 bashname 类似。
path.basename("./test") // test
// 返回路径中文件的后缀名，即路径中最后一个'.'之后的部分。如果一个路径中并不包含'.'或该路径只包含一个'.' 且这个'.'为路径的第一个字符，则此命令返回空字符串。
path.normalize("./test") // test
// 返回路径字符串的对象。
path.parse("./test.js") // { root: '', dir: '.', base: 'test.js', ext: '.js', name: 'test' }
// 从对象中返回路径字符串，和 path.parse 相反。
path.format({ root: "", dir: ".", base: "test.js", ext: ".js", name: "test" }) // ./test.js

// 平台的文件路径分隔符，'\\' 或 '/'。
path.sep // /
// 平台的分隔符, ; or ':'.
path.delimiter // :
// 提供上述 path 的方法，不过总是以 posix 兼容的方式交互。
path.posix
// 提供上述 path 的方法，不过总是以 win32 兼容的方式交互。
path.win32
